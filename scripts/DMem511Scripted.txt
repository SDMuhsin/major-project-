                fifoOut[i][j] = 0;
           end
        end
    end
    else begin
        // Set (i,j)th value = (i,j-1)th value

            for(j= c-1; j > 0; j=j-1)begin
                fifoOut[i][j] =  fifoOut[i][j-1];

        end
        // Load Inputs
        for(i = r-1; i > -1; i=i-1) begin
